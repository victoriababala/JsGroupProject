// Function to render the standings data to the page
function renderStandings(data) {
  const standingsTableBody = document.getElementById('standings');
  data.forEach(team => {
    const row = document.createElement('tr');
    row.innerHTML = `
      <td>${team.rank}</td>
      <td>${team.points}</td>
      <td>${team.name}</td>
      <td><img class="team-logo" src="${team['logo-small']}" alt="${team.name} logo"></td>
      <td><img class="car-image" src="${team['image-car']}" alt="${team.name} car"></td>
    `;
    standingsTableBody.appendChild(row);
  });
}
